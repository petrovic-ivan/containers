# eksctl commands

1. eksctl create cluster <params>                 
# Creating kube cluster with adding inline params

2. eksctl create cluster -f <yaml-file>           
# Create cluster from yaml file

3. eksctl scale nodegroup --cluster=<cluster-name> --nodes=<amount> --name=<group-name>    
# Scale node group to specific amount of nodes

4. eksctl create nodegroup --config-file=<yaml-file> --include=<group-name>                
# Scale node group to specific amount of nodes from yaml file

5. eksctl delete nodegroup --config-file=<yaml-file> --include=<group-name> --approve      
# Deleting node

6. eksctl utils update-cluster-logging --config-file <yaml-file> --approve                 
# Update cluster logging e.g. cloud watch

eksctl get nodegroup --cluster <cluster-name>
